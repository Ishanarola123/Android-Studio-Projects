  for(k=1;k<=count;k++)
  {
   printf(" ");
  }
  count--;
  for(j=1;j<=i;j++)
  {
   printf("*");
  }
  printf("\n");
 }  //end of first for loop
 count=1;
 for(i=2*(number)-1;i>=1;i-=2)//second for loop

   if(i!=(2*(number)-1))
   {
     for(k=1;k<=count;k++)
     {
      printf(" ");
     }
     count++;
     for(j=i;j>=1;j--)
     {
       printf("*");
     }
     printf("\n");

}
