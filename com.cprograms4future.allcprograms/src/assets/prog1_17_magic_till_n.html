int GetSumOfDigits(int n)
{
 int sum=0,x;
 while(n>0)
    {
      x=n%10;
      sum=sum+x;
      n=n/10;
    }
    return sum;
}

int GetReverseOfNumber(int n)
{
 int rev=0,x;
  while(n>0)
     {
       x=n%10;
       rev=rev*10+x;
       n=n/10;
      }
      return rev;
}
