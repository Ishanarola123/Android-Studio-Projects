Question:

What is recursion?

Answer:
Recursion means calling the function by itself.
Ex:
#include<stdio.h>
int factorial(int);
void main()
{
  int value;
  printf(“Enter a value to obtain it’s factorial:”);
  scanf(“%d”,&value);
  int result=factorial(value);
  printf(“factorial of %d is %d”,value,result);
}
int factorial(int f)
{
  if(f!=0)
    return(f*factorial(f-1));
  else
    return 1;
}
