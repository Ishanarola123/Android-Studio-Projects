Question:

What is calling function, called function and function declaration?

Answer:
#include<stdio.h>
void sum();//function declaration
void main()
{
    sum(); //calling function
}
void sum() //called function
{
	int a,b,result;
	printf(“Enter 2 values:”);
	scanf(“%d%d”,&a,&b);
	result=a+b;
	printf(“Sum of %d and %d is %d”,a,b,result);
}

Another Program:

#include<stdio.h>
int sum(int,int);//function declaration
void main()
{
 int a,b,result;
 printf("Enter 2 values:");
 scanf("%d%d",&a,&b);
 result=sum(a,b);//calling function
 printf("Sum of %d and %d is %d",a,b,result);
}

int sum(int a,int b)//called function
{
 int c;
 c=a+b;
 return(c);
}
