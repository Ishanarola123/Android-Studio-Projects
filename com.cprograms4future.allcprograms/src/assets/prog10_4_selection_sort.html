void selectionsort(int a[],int size)
{
 int temp,i,j,pointer;
 for(i=0;i<size-1;i++)
 {
  pointer=i;
  for(j=i+1;j<size;j++)
  {
   if(a[pointer]>a[j])//For descending--> change > to <
   {
    pointer=j;
   }
   if(pointer!=i)
   {
    temp=a[pointer];
    a[pointer]=a[i];
    a[i]=temp;
   }
  }
 }
}
