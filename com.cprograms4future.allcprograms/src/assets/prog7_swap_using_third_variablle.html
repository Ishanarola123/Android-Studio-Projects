main()
{

 scanf("%d",&a);
 printf("Enter b\n");
 scanf("%d",&b);
 printf("The values of a is %d 
     and b is %d before swaping\n",a,b);
 c=a;
 a=b;
 b=c;

       and b is %d after swaping\n",a,b);
}
