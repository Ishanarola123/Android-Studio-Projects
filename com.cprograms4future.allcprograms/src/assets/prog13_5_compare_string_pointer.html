main()
{

int cmp=0;

  scanf("%s",&s1);
  printf("Enter a string 2\n");
  scanf("%s",&s2);
  t1=s1;//copying base address of string
  t2=s2;
  while(*t1!='\0'&&*t2!='\0')
  {
    if(*t1!=*t2)
    {
     cmp++;
    }
    t1++;
    t2++;
  }

  }
  else

   printf("Both are not equal\n");
  }
}
